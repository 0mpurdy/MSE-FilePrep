Sub E_jnd()
    Selection.Find.ClearFormatting
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = "^p<"
        .Replacement.Text = "^p#"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = False
        .MatchCase = True
        .MatchWholeWord = False
        .MatchWildcards = False
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "#[0-9][0-9][0-9][0-9][0-9][EF]*\>"
        .Replacement.Text = "#"
        .MatchWildcards = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "# "
        .Replacement.Text = "#"
        .MatchWildcards = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "{"
        .Replacement.Text = "€"
        .MatchWildcards = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "}"
        .Replacement.Text = "¦"
        .MatchWildcards = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "#"
        .Replacement.Text = ""
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    Selection.EndKey Unit:=wdStory

    strJNDCurrentPage = 384
    While (strJNDCurrentPage > 0)
        strLookForNext = "^p" & strJNDCurrentPage & " "
        strReplaceNext = "{" & strJNDCurrentPage & "}"
    
        With Selection.Find
            .Text = strLookForNext
            .Replacement.Text = "^p"
            .Forward = False
        End With
        Selection.Find.Execute Replace:=wdReplaceOne
    
        Selection.MoveLeft Unit:=wdCharacter, Count:=1
        Selection.MoveRight Unit:=wdCharacter, Count:=1
        Selection.TypeText Text:=strReplaceNext
        Selection.TypeParagraph
        Selection.MoveUp Unit:=wdLine, Count:=1
        strJNDCurrentPage = strJNDCurrentPage - 1
    Wend

    Selection.HomeKey Unit:=wdStory

    With Selection.Find
        .Text = "****"
        .Replacement.Text = "+"
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "*****"
        .Replacement.Text = "+"
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "***"
        .Replacement.Text = "+"
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "**"
        .Replacement.Text = "+"
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll

    With Selection.Find
        .Text = "*"
        .Replacement.Text = "+"
        .Forward = True
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
End Sub

Sub B_addStar()
    'alt-b
    Selection.MoveLeft Unit:=wdCharacter, Count:=1
    Selection.TypeText Text:="*"
    Selection.Find.ClearFormatting
    Selection.Find.Font.Italic = False
    With Selection.Find
        .Text = "^?^?"
        .Replacement.Text = " "
        .Forward = True
        .Wrap = wdFindContinue
        .Format = True
        .MatchCase = False
        .MatchWholeWord = False
        .MatchWildcards = False
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
    Selection.Find.Execute
    Selection.MoveLeft Unit:=wdCharacter, Count:=1
    
    SelEnd = Selection.Range.Start
    Selection.SetRange Start:=SelEnd - 1, End:=SelEnd
    LastChar = Selection.Text
    While LastChar = " " Or LastChar = "." Or LastChar = "," Or LastChar = """" Or LastChar = "'" Or LastChar = "?" Or LastChar = "!" Or LastChar = "-" Or LastChar = ";" Or LastChar = Chr$(13)
       Selection.MoveLeft Unit:=wdCharacter, Count:=1
       SelEnd = Selection.Range.Start
       Selection.SetRange Start:=SelEnd - 1, End:=SelEnd
       LastChar = Selection.Text
    Wend
    
    Selection.MoveRight Unit:=wdCharacter, Count:=1
    Selection.TypeText Text:="*"
    Selection.MoveRight Unit:=wdCharacter, Count:=2
End Sub

Sub M_skipItalics()
    'alt-m
    Selection.Find.ClearFormatting
    Selection.Find.Font.Italic = False
    With Selection.Find
        .Text = "^?^?"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = True
    End With
    Selection.Find.Execute
    Selection.Find.ClearFormatting
    Selection.Find.Font.Italic = True
    With Selection.Find
        .Text = "^?"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = True
    End With
    Selection.Find.Execute
    SelStart = Selection.Range.Start
    Selection.SetRange Start:=SelStart, End:=SelStart + 5
    RemOrQues = Selection.Text
    If RemOrQues = "Rem. " Or RemOrQues = "Ques." Then
        M_skipItalics
    End If
End Sub
